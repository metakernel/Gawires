5/// Enum of all possible subcommands
6#[derive(Debug, PartialEq,Subcommand)]
7    pub enum Commands{
8        Add(Add), // Add new assets or changes to the workpod, add tags and other operations.
9        Checkout(Checkout), // Checkout contents from a remote workpod.
10        Init(Init), // Initialize a connection to a remote project or workpod(For a centralized workflow).
11        Sync(Sync), // Sync the workpod with the distant repository (Will Pull & Push for a distributed workflow).
12        Clone(Clone), // Clone a remote workpod locally, it will clone a complete copy of it.
13        Status(Status), // Display the status of the local workpod
14        Layout(Layout), // Change or create content layout for the workpod (This enables to switch between different layout for the same workpod)
15        Push(Push), // Push the local workpod changes to the distant repository
16        Pull(Pull), // Pull the distant repository changes to the local workpod
17        Workpod(Workpod), // Display the workpod information or create/change to another workpod.
18        Workspace(Workspace), // Display the project workspaces.
19        Preview(Preview), // Preview and compare different commit or assets
20        Reset(Reset), // Reset objects in local workpod to a specific commit (If non use the latest), can use --preserve to keep the local changes or --overwrite to delete them.
21        Remove(Remove), // Remove a file or folder from the local workpod
22        Branch(Branch), // Manage branches inside the workspace
23        Wire(Wire), // Commands related to Wires creation and configuration. Wires are bound to assets and can be used to specify how those assets are handled.
24        Connect(Connect), // Start a LiveWire connection beetween local and remote workpod.
25        Filter(Filter), // Filter files or folders in the workpod or project and return the filtered list
26        Rebase(Rebase), // Rebase a workpod branch on another branch
27        Merge(Merge), // Merge a workpod branch on another branch
28        Clean(Clean), // Clean the workpod
29        Project(Project),
30        Install(Install),
31        Uninstall(Uninstall),
32        Asset(Asset),
33        
34    }

53    /// Checkout assets in local workpod from a source like . When assets are checkout, they are locked by default when in centralized mode.
54    #[derive(Debug, PartialEq, Parser)]
55    pub struct Checkout{
56        /// Specific assets or complete directory structure can be checkout
57        pub path: Option<std::path::PathBuf>,
58        /// Option use to specify if a lock should be put on the assets, this will tell central to lock them. (On by default if centralized, WARNING: If not set to lock you should clone the workpod instead "gawires checkout --clone" if you dont want any conflict issues)
59        #[clap(name = "Lock",short = 'l', long = "lock")]
60        pub lock_remote: Option<bool>,
61        /// Can be use to specify that the checkout must clone the assets instead of synchronizing them.(Will need to initiate a push request each time you try to synchronize)
62        pub clone: Option<bool>,
63
